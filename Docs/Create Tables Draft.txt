CREATE TABLE COURSES (
    CRN SMALLINT,
    Term VARCHAR(20),
    Section VARCHAR(10),
    CatID VARCHAR(10),
    FID VARCHAR(10),
    Credits TINYINT,
    Section VARCHAR(10),
    Capacity SMALLINT,
    Actual SMALLINT,
    Remaining SMALLINT,
    PRIMARY KEY (CRN, Term)
);

CREATE TABLE Course_Meeting (
    MID SMALLINT PRIMARY KEY,
    CRN SMALLINT,
    Term VARCHAR(20),
    LID SMALLINT,
    Day TEXT NOT NULL,
    Start_Time TIME,
    End_Time TIME  
);

CREATE TABLE Catalog_ID (
    Cat_ID VARCHAR(10) PRIMARY KEY,
    Prg_Code VARCHAR(5),
    Title VARCHAR(150),
    Prerequisites VARCHAR(10),
    Corequisites VARCHAR(10)
);

CREATE TABLE Location (
    LID SMALLINT PRIMARY KEY,
    Room VARCHAR(20)
);

CREATE TABLE Faculty (
    FID SMALLINT PRIMARY KEY,
    Faculty_Name VARCHAR(50)
);

CREATE TABLE Program (
    Prg_Code VARCHAR(5),
    Program_Name VARCHAR(150)
);

CREATE TABLE Fees (
    CatID VARCHAR(10),
    Term VARCHAR(20),
    Fee DECIMAL,
    PRIMARY KEY (CatID, Term)
);
